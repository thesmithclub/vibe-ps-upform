function doPost(e) {
  return receiveReservation_(e);
}

function doGet(e) {
  if (e && e.parameter && e.parameter.payload) {
    return receiveReservation_(e);
  }

  return createResponse_({ ok: true, message: '폴스타 신청페이지 접수 Apps Script가 실행 중입니다.' });
}

function receiveReservation_(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const payload = parsePayload_(e);
    const appended = appendReservation_(payload);
    return createResponse_({ ok: true, saved: true, duplicate: !appended });
  } catch (error) {
    return createResponse_({ ok: false, error: error.message });
  } finally {
    lock.releaseLock();
  }
}

function parsePayload_(e) {
  if (e && e.postData && e.postData.contents) {
    try {
      return JSON.parse(e.postData.contents);
    } catch (error) {
      throw new Error('JSON 파싱 실패: ' + error.message);
    }
  }

  if (e && e.parameter && e.parameter.payload) {
    try {
      return JSON.parse(e.parameter.payload);
    } catch (error) {
      throw new Error('GET payload JSON 파싱 실패: ' + error.message);
    }
  }

  return e && e.parameter ? e.parameter : {};
}

function appendReservation_(payload) {
  const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = spreadsheet.getSheetByName(SHEET_NAME) || spreadsheet.insertSheet(SHEET_NAME);
  ensureHeaders_(sheet);

  if (isDuplicateSubmission_(sheet, payload)) {
    return false;
  }

  const row = HEADERS.map(function ([, key]) {
    if (key === '__serverTimestamp') return new Date();
    return normalizeValue_(payload[key]);
  });

  sheet.appendRow(row);
  return true;
}

function isDuplicateSubmission_(sheet, payload) {
  const submissionId = payload && payload.submissionId;
  if (!submissionId) return false;

  const headerLabels = HEADERS.map(function ([label]) { return label; });
  const submissionIdColumn = headerLabels.indexOf('접수 ID') + 1;
  const lastRow = sheet.getLastRow();
  if (submissionIdColumn < 1 || lastRow < 2) return false;

  const values = sheet
    .getRange(2, submissionIdColumn, lastRow - 1, 1)
    .getValues()
    .map(function ([value]) { return String(value); });

  return values.indexOf(String(submissionId)) !== -1;
}

function ensureHeaders_(sheet) {
  const headers = HEADERS.map(function ([label]) { return label; });
  const range = sheet.getRange(1, 1, 1, headers.length);
  const current = range.getValues()[0];
  const hasHeaders = current.some(function (value) { return value !== ''; });

  if (!hasHeaders) {
    range.setValues([headers]);
    sheet.setFrozenRows(1);
    return;
  }

  const isDifferent = headers.some(function (label, index) {
    return current[index] !== label;
  });

  if (isDifferent) {
    range.setValues([headers]);
    sheet.setFrozenRows(1);
  }
}

function normalizeValue_(value) {
  if (value === undefined || value === null) return '';
  if (Array.isArray(value)) return value.join(', ');
  if (typeof value === 'object') return JSON.stringify(value);
  return value;
}

function createResponse_(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
