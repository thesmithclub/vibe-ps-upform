function doPost(e) {
  return receiveReservation_(e);
}

function doGet(e) {
  if (e && e.parameter && e.parameter.payload) {
    return receiveReservation_(e);
  }

  return createResponse_({ ok: true, message: '폴스타 신청페이지 접수 Apps Script가 실행 중입니다.' });
}

function receiveReservation_(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);

  try {
    const payload = parsePayload_(e);
    const appended = appendReservation_(payload);
    return createResponse_({ ok: true, saved: true, duplicate: !appended });
  } catch (error) {
    return createResponse_({ ok: false, error: error.message });
  } finally {
    lock.releaseLock();
  }
}

function parsePayload_(e) {
  if (e && e.postData && e.postData.contents) {
    try {
      return JSON.parse(e.postData.contents);
    } catch (error) {
      throw new Error('JSON 파싱 실패: ' + error.message);
    }
  }

  if (e && e.parameter && e.parameter.payload) {
    try {
      return JSON.parse(e.parameter.payload);
    } catch (error) {
      throw new Error('GET payload JSON 파싱 실패: ' + error.message);
    }
  }

  return e && e.parameter ? e.parameter : {};
}

function appendReservation_(payload) {
  const normalizedPayload = enrichReservationPayload_(payload || {});
  const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = spreadsheet.getSheetByName(SHEET_NAME) || spreadsheet.insertSheet(SHEET_NAME);
  ensureHeaders_(sheet);

  if (isDuplicateSubmission_(sheet, normalizedPayload)) {
    return false;
  }

  const row = HEADERS.map(function ([, key]) {
    if (key === '__serverTimestamp') return new Date();
    return normalizeValue_(normalizedPayload[key]);
  });

  const targetRow = findFirstEmptyRow_(sheet);
  sheet.getRange(targetRow, 1, 1, row.length).setValues([row]);
  return true;
}

function enrichReservationPayload_(payload) {
  const enriched = Object.assign({}, payload);
  enriched.frontFilmSummary = formatPrefixedFilmSummary_('전면', payload.frontFilm);
  enriched.sideOneFilmSummary = formatPrefixedFilmSummary_('측면 1열', payload.sideOneFilm);
  enriched.sideTwoFilmSummary = formatPrefixedFilmSummary_('측면 2열', payload.sideTwoFilm);
  enriched.rearFilmSummary = formatRearFilmSummary_(payload.rearFilm);
  enriched.glassRoofFilmSummary = formatGlassRoofSummary_(payload.glassRoofFilm);
  enriched.blackboxSummary = formatBlackboxSummary_(payload.blackbox, payload.blackboxChannel);
  enriched.ppfSummary = formatPpfSummary_(payload.ppf);
  enriched.hipassSummary = formatHipassSummary_(payload.hipass);
  return enriched;
}

function formatPrefixedFilmSummary_(label, value) {
  const text = cleanText_(value);
  return text ? label + ' : ' + text : label + ' : 선택하지 않음';
}

function formatRearFilmSummary_(value) {
  const text = cleanText_(value);
  if (!text || text.indexOf('후면 유리가 없습니다') !== -1 || text === '해당사항 없음') {
    return '해당사항 없음';
  }
  return '후면 : ' + text;
}

function formatGlassRoofSummary_(value) {
  const text = cleanText_(value);
  if (!text || text === '선택하지 않음') return '선택하지 않음';
  return text;
}

function formatBlackboxSummary_(blackbox, blackboxChannel) {
  if (!isSelected_(blackbox)) return '선택하지 않음';
  const channel = cleanText_(blackboxChannel);
  return channel || '채널 미선택';
}

function formatPpfSummary_(value) {
  return isSelected_(value) ? '3종 선택' : '선택하지 않음';
}

function formatHipassSummary_(value) {
  return isSelected_(value) ? '매립 요청' : '선택하지 않음';
}

function cleanText_(value) {
  return String(normalizeValue_(value)).trim();
}

function isSelected_(value) {
  const text = cleanText_(value).toLowerCase();
  return value === true || text === 'true' || text === '예' || text === 'yes' || text === 'y';
}

function findFirstEmptyRow_(sheet) {
  const startRow = 2;
  const maxRows = sheet.getMaxRows();
  const columnCount = HEADERS.length;

  if (maxRows < startRow) {
    sheet.insertRowsAfter(maxRows, startRow - maxRows);
  }

  const rowCount = Math.max(sheet.getMaxRows() - startRow + 1, 1);
  const values = sheet.getRange(startRow, 1, rowCount, columnCount).getValues();
  const emptyIndex = values.findIndex(function (row) {
    return row.every(function (value) {
      return value === '' || value === null;
    });
  });

  if (emptyIndex !== -1) {
    return startRow + emptyIndex;
  }

  sheet.insertRowAfter(sheet.getMaxRows());
  return sheet.getMaxRows();
}

function isDuplicateSubmission_(sheet, payload) {
  const submissionId = payload && payload.submissionId;
  if (!submissionId) return false;

  const headerLabels = HEADERS.map(function ([label]) { return label; });
  const submissionIdColumn = headerLabels.indexOf('접수 ID') + 1;
  const lastRow = sheet.getLastRow();
  if (submissionIdColumn < 1 || lastRow < 2) return false;

  const values = sheet
    .getRange(2, submissionIdColumn, lastRow - 1, 1)
    .getValues()
    .map(function ([value]) { return String(value); });

  return values.indexOf(String(submissionId)) !== -1;
}

function ensureHeaders_(sheet) {
  const headers = HEADERS.map(function ([label]) { return label; });
  const range = sheet.getRange(1, 1, 1, headers.length);
  const current = range.getValues()[0];
  const hasHeaders = current.some(function (value) { return value !== ''; });

  if (!hasHeaders) {
    range.setValues([headers]);
    sheet.setFrozenRows(1);
    return;
  }

  const isDifferent = headers.some(function (label, index) {
    return current[index] !== label;
  });

  if (isDifferent) {
    range.setValues([headers]);
    sheet.setFrozenRows(1);
  }
}

function normalizeValue_(value) {
  if (value === undefined || value === null) return '';
  if (Array.isArray(value)) return value.join(', ');
  if (typeof value === 'object') return JSON.stringify(value);
  return value;
}

function createResponse_(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
